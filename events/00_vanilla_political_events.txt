namespace = vanilla_political_events

# Democratic Election Event
country_event = {
    id = vanilla_political_events.1
    title = playabletwr_democratic_election_title
    desc = playabletwr_democratic_election_desc
    picture = GFX_report_event_democracy
    trigger = { has_government = democratic }
    option = {
        name = playabletwr_democratic_election_option
        set_politics = { ruling_party = democratic elections_allowed = yes }
    }
}

# Civil War Event
country_event = {
    id = vanilla_political_events.2
    title = playabletwr_civil_war_title
    desc = playabletwr_civil_war_desc
    picture = GFX_report_event_civil_war
    trigger = { NOT = { has_government = democratic } }
    option = {
        name = playabletwr_civil_war_option
        start_civil_war = { ideology = democratic size = 0.3 }
    }
}
